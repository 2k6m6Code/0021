Add One to One Mapping in Batch Way

where MAPPING_n:= { FORMAT1 | FORMAT2 | FORMAT3 }

FORMAT1 := virtual_ip > real_server

FORMAT2 := start virtual_ip : number of ip > start real_server

FORMAT3 := start virtual_ip - end virtual_ip > start real_server
